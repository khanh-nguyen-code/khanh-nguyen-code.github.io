from typing import Callable


def f(n: int) -> int:
    def _log2(n: int) -> int:
        return len(bin(n)) - 3

    def _find_i(n: int) -> int:
        if n < 4:
            # S_0 and S_1
            return 1
        else:
            # S_2, S_3, ...
            return _log2(n)

    def f_(i: int, n: int) -> int:
        if n == 2 ** (i + 1) - 1:
            # max element
            if i == 1:
                # S_0 and S_1
                return 1
            else:
                # S_2, S_3, ...
                return 2 ** i
        else:
            # other element
            return n + 1

    i = _find_i(n)
    return f_(i, n)


def c_min(f: Callable[[int], int], n: int) -> int:
    m = 1
    out = f(n)
    while True:
        m += 1
        out = f(out)
        if out == n:
            return m


if __name__ == "__main__":
    for n in range(1, 1000):
        print(f"{n}: c_min {c_min(f, n)}")
